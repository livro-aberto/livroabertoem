	\teacherenvtrue
	\raggedright
	\la@widelist
	\small
}{
	\par\hfill
	\la@defaultlist
	\teacherenvfalse
	\normalsize
}

	\par\switchcolumn

		}{

	\par\switchcolumn
}

\newenvironment{teacher*}{
	\par\switchcolumn*
	\begin{la@teacher}
		}{
	\end{la@teacher}
	\par\switchcolumn
}

\newcommand{\teachertitle}[3][\currentcolor]{
	\vbox{
    \color{#1}\bfseries\vspace{1em}
    {\Large\selectfont #2}\par
    \vspace{-.8\baselineskip}
    \rule{\la@teacher@textwidth}{3pt}\par
    \vspace{-.6\baselineskip}
    {\large\RaggedRight #3}\par
  }%\normalfont\small\color{black}
}

\newcommand{\la@teacher@sidetitle}[2]{%
    \vspace{.5\la@titlebeforeskip}
	\ifteacherenv\penalty-1000\par
    \teachertitle{#1}{#2}
    \penalty10000\par
    \vspace{.5\la@titleafterskip}
	\else
		\la@notteacherenv
	\fi
}

\newenvironment{objectives}[1]{
	\la@teacher@sidetitle{Objetivos Específicos}{#1}
}{}

}{}
