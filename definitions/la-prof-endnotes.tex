\newlength{\teachernoteafterskip}
\setlength{\teachernoteafterskip}{\la@titlebeforeskip}
\newcommand{\teachernotelist}{\la@widelist}
\newcommand{\la@teacher@note}[3]{
		\leavevmode\pagenote{
      \teachernotelist%
      \teachertitle[\currentvolcolor]{#1}{#2}
      \penalty10000\par #3 \par
      \vspace*{\teachernoteafterskip}
	}
}
