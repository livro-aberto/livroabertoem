% Page number command
\newcommand{\la@evenpagenumber}{%
	\tikzset{external/export next=false}

	\begin{tikzpicture}[remember picture, overlay]

		% \coordinate (bottom) at ($(current page.south west) + (13.50mm,6.80mm)$);

		\begin{pgfonlayer}{main}

			% Curve below
			\fill [fill = principal]
				($(current page.south west) + (0,\la@footerbaseline)$) 
				-- ++(0,-.5\la@footerheight) -- ++(\la@footershortwidth,0)
				.. controls +(\la@footerctrla,0) and +(\la@footerctrlb,.2\la@footerheight) ..
				++ (\la@footerlongwidth, .5\la@footerheight) -- 
				cycle;

			% Curve below
			\fill [fill = principal]
				($(current page.south west) + (0,\la@footerbaseline)$) 
				-- ++(0,+.5\la@footerheight) -- ++(\la@footershortwidth,0)
				.. controls +(\la@footerctrla,0) and +(\la@footerctrlb, -.2\la@footerheight) ..
				++ (\la@footerlongwidth, -.5\la@footerheight) -- 
				cycle;

			\draw [principal, line width = 3pt] 
				($(current page.south west) + (0,\la@footerbaseline)$) 
				-- ($(current page.south east) + (-\la@student@foremargin,\la@footerbaseline)$);

      \node [anchor = east] at ($(current page.south west) + (.8\la@student@foremargin,\la@footerbaseline)$)  {\footerfont\thepage};
				
								
		\end{pgfonlayer}

		% \begin{pgfonlayer}{foreground}
		% 	\node [text=white, font=\bfseries](page_number) at ($(bottom)!0.5!(corner)$) {\thepage};
		% \end{pgfonlayer}
	\end{tikzpicture}
}

				bottom=-2pt, 
				hbox, 
				sharp corners = northeast, 
				sharp corners = southwest
			]{\textbf{#1}}
			\end{tcolorbox}
		}
	}
